int a = 120;
byte b = a + 10;           // Присваивание значения типа int переменной типа byte без явного преобразования вызывает ошибку компиляции, так как типы несовместимы.
byte c = (byte)(a + 10);   // Результат выражения a + 10 равен 130. При преобразовании 130 из int в byte происходит переполнение, так как byte может содержать значения от -128 до 127. Значение 130 преобразуется циклически: 130−256=−126.
byte d = a + 1;            // Присваивание значения типа int переменной типа byte без явного преобразования вызывает ошибку компиляции, так как типы несовместимы.
